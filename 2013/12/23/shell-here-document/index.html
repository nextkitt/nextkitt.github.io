shell - here文档

2013.12.23 nextkitt Linux  热度 ℃

here文档的作用：

  1. 替换（或不替换）变量和命令
  2. 多行数据格式化输出（比使用echo或printf更易读）
  3. 多行注释

实例代码如下：

#!/bin/bash
function _test()
{
    var1="Hello";
    var2="World";

    # 变量或命令的替换
    cat <<EOF
$var1 $var2 redirect to stdout
EOF

    echo "--- --- ---";

    # 变量或命令的不替换
    cat <<"EOF"
$var1 $var2 redirect to stdout
EOF

    echo "--- --- ---";

    # 多行数据格式化到文件
    cat > file <<EOF
$var1 $var2 redirect to file
EOF

    echo "--- --- ---";

    # 多行数据作为命令的输入
    tr a-z A-Z <<EOF
tr $var1 $var2
EOF

    echo "--- --- ---";

    #忽略行首的tab字符 [测试失败]
    cat <<-EOF
    $var1 $var2 ignore tab
EOF
}

_test;
